CREATE OR REPLACE Function get_nextcode(p_table_name VARCHAR,p_col_name VARCHAR,p_length NUMBER) return VARCHAR is
  v_query VARCHAR(200);
  v_autocode VARCHAR(50);
Begin
 v_query := 'select lpad(nvl(max(to_number('||p_col_name||'))+1,1),'||p_length||',0) code from '||p_table_name;
 EXECUTE IMMEDIATE v_query INTO v_autocode;
 return v_autocode;
End;
/

CREATE OR REPLACE Function get_nextcode_gby(p_table_name VARCHAR,p_col_name VARCHAR,p_length NUMBER,p_group_by VARCHAR) return VARCHAR is
   v_group_by_length NUMBER;
   v_autocode  VARCHAR(50);
   v_query VARCHAR(300);
Begin
   v_group_by_length := length(p_group_by);

   v_query := 'Select Lpad(Nvl(Max(To_Number(substr('||p_col_name||','||to_number(v_group_by_length+1)||')))+1,1),'||To_number(p_length-v_group_by_length)||',0) From '||p_table_name||' Where substr('||p_col_name||',1,'||v_group_by_length||')='''||p_group_by||'''';

   EXECUTE IMMEDIATE v_query INTO v_autocode;
   return (p_group_by||v_autocode);
End;
/

CREATE OR REPLACE Function date_format(p_format VARCHAR) return VARCHAR is
   v_format VARCHAR(30);
Begin
   Select To_Char(Sysdate,p_format) INTO v_format from Dual;
   return Trim(v_format);
End;

CREATE OR REPLACE Function getAccHeadOpenBal(p_acchead VARCHAR,p_trust VARCHAR,p_date date,p_param varchar ) return VARCHAR is
   openbalance number(15,4);
Begin
   select decode(p_param,'C',sum(credit),'D',sum(debit),(sum(credit) - sum(debit))) into openbalance
  from (select Nvl(decode(amounttype, 'CR', amount, 'DR', 0), 0) credit,
               Nvl(decode(amounttype, 'CR', 0, 'DR', amount), 0) debit
          from cb_accountcode_info info, cb_accountcode_details dt
         where TRUSTTYPE = p_trust and
               info.accounthead = dt.accounthead and info.accounthead = p_acchead and
               info.opendate <= decode(p_date, '', '01/Jan/1000', p_date)
        union all
        select sum(credit) credit,
               sum(debit) debit
          from Cb_VOUCHER_INFO     info,
               Cb_voucher_details  dt,
               cb_accountcode_info ainfo
         where dt.keyno = info.keyno and ainfo.accounthead = dt.accounthead and
              TRUSTTYPE = p_trust and
               info.voucher_dt < decode(p_date, '', '01/Jan/1000', p_date) and
               info.voucher_dt >=
               decode(ainfo.opendate, '', '01/Jan/1000', ainfo.opendate) and ainfo.accounthead = p_acchead
         group by dt.accounthead, info.trusttype
        union all
        select sum(credit) credit,
               sum(debit) debit
          from Cb_Journalvoucher         info,
               Cb_Journalvoucher_Details dt,
               cb_accountcode_info       ainfo
         where dt.keyno = info.keyno and ainfo.accounthead = dt.ACCOUNTCODE and
               TRUSTTYPE = p_trust and
               info.voucher_dt < decode(p_date, '', '02/Apr/2011', p_date) and
               info.voucher_dt >= decode(ainfo.opendate, '', '01/Jan/1000', ainfo.opendate) and ainfo.accounthead = p_acchead
         group by info.trusttype, ACCOUNTCODE) ;
   return openbalance;
End;
/
